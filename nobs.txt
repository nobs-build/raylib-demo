unit demo

build: +bckg_res +music_res logo_res
{
  OUTPUT_NAME="demo${OUTPUT_SUFFIX}"
  C_FILES = P"(demo.c)"
  C_FLAGS = [ "-Os", "-ffunction-sections", "-fdata-sections" ]
  LINK_FLAGS = [ "--gc-sections", "--strip-all" ]
  OBJS += logo_res.OBJS
  LIB = [ "-lraylib" ]
  if os_family = "windows"
  {
    LIB += [ "-lgdi32", "-lwinmm" ]
  }
  if os_family = "unix"
  {
    LIB += [ "-lGL", "-lm", "-lpthread", "-ldl", "-lrt", "-lX11" ]
  }
}

bckg_res:
{
  ASSET_RESOURCE
  ASSET_INPUT = P"(bckg.png)"
  // This variable is for cache freshness only. Rebuild the generated object when the asset helper scripts change.
  CACHE_FRESHNESS = [ P"(tools/Optimize-Asset.${ASSET_SCRIPT_EXT})", P"(tools/Build-AssetResource.${ASSET_SCRIPT_EXT})" ]
  ^OBJS = [ P"(out/generated-assets/bckg.o)" ]
}

music_res:
{
  ASSET_RESOURCE
  ASSET_INPUT = P"(braccada_jam.ogg)"
  // This variable is for cache freshness only. Rebuild the generated object when the asset helper scripts change.
  CACHE_FRESHNESS = [ P"(tools/Optimize-Asset.${ASSET_SCRIPT_EXT})", P"(tools/Build-AssetResource.${ASSET_SCRIPT_EXT})" ]
  ^OBJS = [ P"(out/generated-assets/braccada_jam.o)" ]
}

logo_res:
{
  OBJCOPY
  INPUT = P"(logo.png)"
  OBJCOPY_FLAGS = [
    "-I binary",
    // Some objcopy builds do not support --binary-symbol-prefix logo.
    // "--binary-symbol-prefix logo",
    "--rename-section .data=logo,alloc,load,readonly,data,contents",
    "-O ${OBJCOPY_FORMAT}",
    "-B i386:x86-64"
  ]
  OUTPUT_DIR = P"(out/generated-assets)"
  OUTPUT_NAME = "logo.o"
  ^OBJS = [ P"(out/generated-assets/logo.o)" ]
}
